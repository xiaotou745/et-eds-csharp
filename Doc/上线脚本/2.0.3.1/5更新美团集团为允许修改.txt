

use superman
go
declare @countrow INT
  --受影响行数
set @countrow = 1  --预计受影响行数

begin transaction
--更新商家所属集团是否允许修改
update dbo.[group] set IsModifyBind=1 where id =4
if ( @@error <> 0
     or @@rowcount <> @countrow
   ) 
    begin
        rollback
        print 'RollBack'
        return
    end

print 'Commit Start'
commit
print 'Commit Over'
