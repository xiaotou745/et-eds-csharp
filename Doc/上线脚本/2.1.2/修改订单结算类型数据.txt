
 use superman
 go
 ------------1结算比例 修改固定金额为0
 declare @countrow int;
  --受影响行数 
 select @countrow = count(1) FROM dbo.[order] (nolock)
where CommissionType = 1
 begin tran
update dbo.[order]
set CommissionFixValue=0
where CommissionType=1
 if ( @@error <> 0
      or @@rowcount <> @countrow
    )
    begin
        rollback;
        print 'RollBack';
        return;
    end; 
 print 'Commit Start';
 commit;
 print 'Commit Over'; 
go
